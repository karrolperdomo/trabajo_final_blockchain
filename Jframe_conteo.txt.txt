    public void cargarDatos(){
        ManejoVotos  objManejoVotos= new ManejoVotos();
        
        int[] conteo = objManejoVotos.conteoVotos();
        String[] candidatos = objManejoVotos.getCandidatos();
        
        
        DefaultTableModel model = new DefaultTableModel();
        model.addColumn("Nombre Candidato");
        model.addColumn("Cantidad de Votos");
        
        for(int i= 0; i < candidatos.length; i++){
            String columnas[] = {candidatos[i],conteo[i] + "" };
            model.addRow(columnas);
        }
        jTable1.setModel(model);
    }
