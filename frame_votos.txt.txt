    private void jButton_AlejoActionPerformed(java.awt.event.ActionEvent evt) {                                              
        // TODO add your handling code here:
        int numeroCandidato = 2;
        String identificacion = jField_identificacion.getText();
        if (identificacion.isEmpty() || identificacion != "" ) {
            JOptionPane.showMessageDialog(null, "Usted voto por Alejandro");
        } else {
            JOptionPane.showMessageDialog(null, "Ingrese su numero de identificacion");
        }
        
        ManejoVotos objManejoVotos = new ManejoVotos();
        objManejoVotos.crearCandidatos();
        boolean estado = objManejoVotos.agregarVoto(identificacion, numeroCandidato);
        objManejoVotos.imprimirVotos();
        if (estado == true) {
            JOptionPane.showMessageDialog(null, "Su voto se registro correctamente");
            jField_identificacion.setText("");
        } else {
            JOptionPane.showMessageDialog(null, "ERROR: Usted ya tiene un registro de voto");
        }
    }
